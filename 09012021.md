```python
class Solution:
    def candy(self, ratings: List[int]) -> int:
        n = len(ratings)
        left = [1] * n
        right = left[:]
        res = 0
        for i in range(1, n):
            if ratings[i] > ratings[i - 1]:
                left[i] = left[i - 1] + 1
        res = left[-1]
        for i in range(n - 2, -1, -1):
            if ratings[i] > ratings[i + 1]:
                right[i] = right[i + 1] + 1
            res += max(left[i], right[i])
        return res
```

```python
        """

        self.right = []

        if len(self.left) == len(self.right):
            heappush(self.left, - heappushpop(self.right, num))

            heappush(self.right, - heappushpop(self.left, - num))
```

```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        issue_nums = []
        prev = None
        def traverse(root):
            nonlocal issue_nums, prev
            if not root:return
            traverse(root.left)
            if prev and prev.val > root.val:
                issue_nums.append([prev, root])
            prev = root
            traverse(root.right)
            return
        traverse(root)
        first = issue_nums[0][0]
        k = int(len(issue_nums) == 2)
        second = issue_nums[k][1]
        first.val,second.val = second.val, first.val
        return
```

```python
class Node:
    def __init__(self, key = 0, value = 0, freq = 0):
        self.key = key
        self.value = value
        self.freq = freq
        self.prev = None
        self.next = None
    def insert(self, node):
        self.next.prev = node
        node.prev = self
        node.next = self.next
        self.next = node
    def __repr__(self):
        return '[ val:' + str(self.value) + ', freq' + str(self.freq) +']'
def init_list():
    head, tail = Node(), Node()
    head.next = head.prev = tail
    tail.next = tail.prev = head
    return [head, tail]
class LFUCache:
    def del_node(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev
        return node.key
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cur_cap = 0
        self.cur_min = 0
        self.freq_lsit = defaultdict(init_list)
        self.key_node = {}
    def update_cur_min(self, node):
        freq = node.freq
        if freq == 1:
            self.cur_min = 1
            return
        if self.cur_min == freq - 1:
            head, tail = self.freq_lsit[freq - 1]
            if head.next == tail:
                self.cur_min = freq
        return
    def update_node(self, node):
        freq = node.freq
        _, tail = self.freq_lsit[freq]
        tail.prev.insert(node)
        self.update_cur_min(node)
        return
    def get(self, key: int) -> int:
        # print('get',self.freq_lsit)
        if key not in self.key_node:return -1
        node = self.key_node[key]
        self.del_node(node)
        node.freq += 1
        self.update_node(node)
        return node.value
    def put(self, key: int, value: int) -> None:
        # print('put',self.freq_lsit)
        if key in self.key_node:
            node = self.key_node[key]
            node.value = value
            node.freq += 1
            self.del_node(node)
            self.update_node(node)
            return
        if self.capacity == 0:return
        if self.capacity == self.cur_cap:
            node = self.freq_lsit[self.cur_min][0].next
            self.del_node(node)
            self.key_node.pop(node.key)
            self.cur_cap -= 1
        node = Node(key, value, 1)
        self.cur_cap += 1
        self.update_node(node)
        self.key_node[key] = node
        return
```

```python
class TreeAncestor:

    def __init__(self, n: int, parent: List[int]):
        k = int(log2(n)) + 1
        self.dp = [[-1] * k for _ in range(n)]
        for i in range(n):
            self.dp[i][0] = parent[i]
        for i in range(n):
            for j in range(1, k):
                if self.dp[i][j - 1] != -1:

            j = int(log2(k))
            node = self.dp[node][j]
            k -= 1 << j
```

```python
class Solution:
    def solveNQueens(self, n: int) -> List[List[str]]:
        res = []
        pos = set()
        neg = set()
        col = set()

        board = [['.'] * n for _ in range(n)]

        def bt(r):
            if r == n:
                copy = [''.join(r) for r in board]
                res.append(copy)
                return
            for c in range(n):
                if c in col or (r - c) in pos or (r + c) in neg:
                    continue
                col.add(c)
                pos.add(r - c)
                neg.add(r + c)
                board[r][c] = 'Q'
                bt(r + 1)
                col.remove(c)
                pos.remove(r - c)
                neg.remove(r + c)
                board[r][c] = '.'
            return
        bt(0)
        return res
```

```python
class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        get = lambda x: mountain_arr.get(x)
        def bs(l, r, cond):
            while l < r:
                mid = l + (r - l) // 2
                if cond(mid):
                    r = mid
                else:
                    l = mid + 1
            return l
        n = mountain_arr.length()
        p = bs(0, n - 1, lambda m: get(m) > get(m + 1))
        l = bs(0, p, lambda m: get(m) >= target)
        if get(l) == target:return l
        r = bs(p, n - 1, lambda m: get(m) <= target)
        if get(r) == target:return r
        return -1
```
